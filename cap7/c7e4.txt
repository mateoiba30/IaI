procesos
  proceso recorrer_calle
  variables
    p:numero
    f:numero
    e:numero
    x:numero
    y:numero
  comenzar
    e:=0
    x:=0
    repetir 99
      p:=0
      f:=0
      y:=0
      si(HayFlorEnLaEsquina)&(HayPapelEnLaEsquina)
        y:=1
      mientras(HayPapelEnLaEsquina)
        tomarPapel
        p:=p+1
      mientras(HayFlorEnLaEsquina)
        tomarFlor
        f:=f+1
      x:=p+p
      si(y=1)&(f=x)
        e:=e+1
      mover
    p:=0
    f:=0
    mientras(HayPapelEnLaEsquina)
      tomarPapel
      p:=p+1
    mientras(HayFlorEnLaEsquina)
      tomarFlor
      f:=f+1
    x:=p+p
    si(y=1)&(f=x)
      e:=e+1
    Informar(e)
  fin
